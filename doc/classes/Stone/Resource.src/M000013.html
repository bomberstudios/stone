    def field(name, klass, arg = nil)
      
      if arg && arg[:unique] == true
        unique = true
      else
        unique = false
      end
      
      klass_sym = self.to_s.make_key
      unless @@fields[klass_sym]
        @@fields[klass_sym] = [{:name => name, 
                                :klass => klass, 
                                :unique => unique}]
      else
        @@fields[klass_sym] << {:name => name, 
                                :klass => klass, 
                                :unique => unique}
      end
      
      name = name.to_s
      
      self.class_eval "def \#{name}\n@\#{name}\nend\n\ndef \#{name}=(value)\n@\#{name} = value\nend\n", __FILE__, __LINE__
    end
