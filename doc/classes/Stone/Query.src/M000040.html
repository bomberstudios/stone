    def initialize(field,op)
      self.field = field
      self.op = case op
                  when :gt
                    ".>"
                  when :lt
                    ".<"
                  when :gte
                    ".>="
                  when :lte
                    ".<="
                  when :includes
                    ".include?"
                  when :matches
                    ".=~"
                  when :equals
                    ".=="
                  when :not
                    ".!="
                  end
    end
