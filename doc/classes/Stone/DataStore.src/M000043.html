      def load_data(sym)
        dir = self.local_dir
        ymls = Dir.glob(dir/sym.to_s.pluralize/"*.yml")
        objs = []
        unless ymls.empty?
          ymls.each do |yml|
            obj = YAML.load_file yml
            objs << [obj.id, obj]
          end
        end
        return objs
      end
